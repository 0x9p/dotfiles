* Org

#+begin_src emacs-lisp

(custom-theme-set-faces!
'doom-one
'(org-level-8 :inherit outline-3 :height 1.0)
'(org-level-7 :inherit outline-3 :height 1.0)
'(org-level-6 :inherit outline-3 :height 1.1)
'(org-Level-5 :inherit outline-3 :height 1.2)
'(org-Level-4 :inherit outline-3 :height 1.3)
'(org-level-3 :inherit outline-3 :height 1.4)
'(org-level-2 :inherit outline-2 :height 1.5)
'(org-level-1 :inherit outline-1 :height 1.6)
'(org-document-title :height 1.8 :bold t :underline nil))

(setq org-agenda-files (directory-files-recursively "~/workspace" "\\.org$"))
(setq org-directory "~/workspace")
(setq org-modern-table-vertical 1)
(setq org-modern-table t)

(after! org
  (setq org-todo-keywords
        '((sequence
           "TODO(t)"
           "IN PROGRESS(p)"
           "IN REVIEW(r)"
           "IN TESTING(s)"
           "BLOCKED(b)"
           "DONE(d)")))

  (setq org-todo-keyword-faces
        '(("TODO"        . (:foreground "#7aa2f7" :weight bold))
          ("IN PROGRESS" . (:foreground "#f78c6c" :weight bold))
          ("IN REVIEW"   . (:foreground "#e0af68" :weight bold))
          ("IN TESTING"  . (:foreground "#73daca" :weight bold))
          ("BLOCKED"     . (:foreground "#f7768e" :weight bold))
          ("DONE"        . (:foreground "#9ece6a" :weight bold)))))

#+end_src
